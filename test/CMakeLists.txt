cmake_minimum_required(VERSION 3.8)
set(This TwitchNetworkTransportTests)

    src/ConnectionTests.cpp
)

    FOLDER Tests
)

    gtest_main
    SystemAbstractions
    TwitchNetworkTransport
)

    NAME ${This}
    COMMAND ${This}
)
